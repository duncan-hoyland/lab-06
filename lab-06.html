gmean <- function(x){
  if(any(!is.numeric(x)) || any(x <= 0)) {
    warning("X inputs must be greater than zero and numerical for geometric mean calculations")
    return(NaN)
  }
  geometric_mean <- exp(mean(log(x)))
  
  return(geometric_mean)
}
